// function functionname(){}

function subtract(a,b){
    return a-b;
}
 function mul(a,b){ 
     // here a and b represents the function parameter..
     // here multiply will be the operation
     // for every function we should have the return keyword...
     return a*b;
 }
 function add(a,b){
     return a+b;
 }
